def show_phase(y, f):
    ff = fft(y)
    ang = np.angle(ff, deg=True)
    ang1 = ang.copy()
    ang1[np.abs(ff)<0.01]=0
    fig, ax = plt.subplots(2)
    for a in ax:
        a.set_xlabel("f[Hz]")
        a.set_ylabel(r"$\varphi$[deg]")
        a.set_yticks(np.arange(-180, 181, 45))
        a.set_xticks(range(n3))
    ax[0].stem(f, ang)
    ax[1].stem(f, ang1)
